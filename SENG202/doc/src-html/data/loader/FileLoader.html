001package data.loader;
002
003import java.io.BufferedReader;
004import java.io.File;
005import java.io.FileInputStream;
006import java.io.FileNotFoundException;
007import java.io.IOException;
008import java.io.InputStream;
009import java.io.InputStreamReader;
010import java.util.ArrayList;
011import java.util.Calendar;
012import java.util.GregorianCalendar;
013
014import data.model.DataPoint;
015import data.model.Event;
016import data.model.EventContainer;
017
018/**
019 * This class provides a way of reading activity events provided by a fitness
020 * tracking device. The activity data needs to be in a CSV format consisting of
021 * date, time, heart rate, latitude, longitude & altitude.
022 * 
023 * @author Fitr.Team
024 */
025public class FileLoader {
026        private InputStream inputStream;
027        private ArrayList<Event> events = new ArrayList<Event>();
028        private EventContainer eventContainer = new EventContainer();
029
030        /**
031         * Creates a FileLoader with a particular input file
032         * 
033         * @param file
034         *            The input file
035         */
036        public FileLoader(File file) {
037                try {
038                        inputStream = new FileInputStream(file);
039                } catch (FileNotFoundException e) {
040                        e.printStackTrace();
041                        clearStream();
042                } catch (NullPointerException e) {
043                        e.printStackTrace();
044                }
045        }
046
047        public FileLoader() {
048                inputStream = this.getClass().getResourceAsStream(
049                                "seng202_2014_example_data.csv");
050        }
051
052        /**
053         * Gets the input stream
054         * 
055         * @return The input stream
056         */
057        public InputStream getStream() {
058                return inputStream;
059        }
060
061        /**
062         * Reads a CSV file and creates a new event at each #start add all following
063         * data points to the event
064         */
065        public void load() {
066                BufferedReader br = null;
067                String line = "";
068                DataPoint lastPoint = null;
069                Event currentEvent = new Event("");
070
071                try {
072                        br = new BufferedReader(new InputStreamReader(inputStream));
073
074                        while ((line = br.readLine()) != null) {
075                                String[] dataLine = line.split(",");
076                                if (isValidLine(line) || line.startsWith("#start")) {
077                                        if (dataLine[0].contains("#start")) {
078                                                currentEvent = new Event(dataLine[1]);
079                                                events.add(currentEvent);
080                                                eventContainer.addEvent(currentEvent); // we should be using EventContainer not an arrayList???
081                                                lastPoint = null;
082                                        } else {
083                                                String[] dateString = dataLine[0].split("/");
084                                                String[] time = dataLine[1].split(":");
085
086                                                // months start from 0...
087                                                Calendar date = new GregorianCalendar(
088                                                                Integer.parseInt(dateString[2]), // Year
089                                                                Integer.parseInt(dateString[1]), // Month
090                                                                Integer.parseInt(dateString[0]), // Day
091                                                                Integer.parseInt(time[0]), // Hour
092                                                                Integer.parseInt(time[1]), // Minute
093                                                                Integer.parseInt(time[2])); // Second
094
095                                                int heartrate = Integer.parseInt(dataLine[2]);
096                                                double latitude = Double.parseDouble(dataLine[3]);
097                                                double longitude = Double.parseDouble(dataLine[4]);
098                                                double altitude = Double.parseDouble(dataLine[5]);
099
100                                                DataPoint point = new DataPoint.Builder().date(date)
101                                                                .heartRate(heartrate).latitude(latitude)
102                                                                .longitude(longitude).altitude(altitude)
103                                                                .prevDataPoint(lastPoint).build();
104
105                                                currentEvent.addDataPoint(point);
106                                                lastPoint = point;
107                                        }
108                                }
109                        }
110
111                } catch (FileNotFoundException e) {
112                        e.printStackTrace();
113                        System.out.println("File Not Found");
114                } catch (IOException e) {
115                        e.printStackTrace();
116                        System.out.println("couldnt read line");
117                }
118
119        }
120
121        public boolean isValidLine(String line) {
122                String dataLine = "(\\d){2}/(\\d){2}/(\\d){4},(\\d){2}:(\\d){2}:(\\d){2}"
123                                + ",(\\-)?(\\d)+.(\\d)+,(\\-)?(\\d)+.(\\d)+,(\\d){2,3}(.(\\d))?";
124                
125                return line.matches(dataLine);
126        }
127
128        public ArrayList<Event> getEvents() {
129                return events;
130        }
131
132        private void clearStream() {
133                inputStream = null;
134        }
135}
