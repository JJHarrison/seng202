144        /**
145         * constructor for DataPoint Set data point values using builder
146         * 
147         * @param builder
148         */
149        public DataPoint(Builder builder) {
150                this.date = builder.date;
151                this.heartRate = builder.heartRate;
152                this.latitude = builder.latitude;
153                this.longitude = builder.longitude;
154                this.altitude = builder.altitude;
155
156                if (builder.previousPointPoint != null) {
157                        this.distance = calculateDistance(builder.previousPointPoint);
158                        this.speed = calculateSpeed(calculateDeltaTime(builder.previousPointPoint));
159                } else {
160                        this.distance = 0.0;
161                        this.speed = 0.0;
162                }
163        }
164
165        /**
166         * Calculates the change in time between two points.
167         * 
168         * @param previousPoint
169         *            The point previous to this point in an event.
170         * @return The change in time (seconds).
171         */
172        private long calculateDeltaTime(DataPoint previousPoint) {
173                long previousTime = previousPoint.getDate().getTimeInMillis();
174                return ((date.getTimeInMillis() - previousTime) / 1000);
175        }
176
177        /**
178         * Calculates the distance from the current point to the previousPoint in
179         * meters.
180         * 
181         * @param previousPoint
182         *            The point from which to calculate the distance from
183         * @return The distance in meters from the previous point
184         */
185        private double calculateDistance(DataPoint previousPoint) {
186                double distance = 0;
187                double radius = 6373 * 1000; // Converted to meters
188                double latPrev = previousPoint.getLatitude();
189                double lonPrev = previousPoint.getLongitude();
190
191                double deltaLat = latPrev - latitude;
192                double deltaLon = lonPrev - longitude;
193
194                double a = Math.pow(Math.sin(Math.toRadians(deltaLat / 2)), 2)
195                                + (Math.cos(Math.toRadians(latitude))
196                                                * Math.cos(Math.toRadians(latPrev)) * Math.pow(
197                                                Math.sin(Math.toRadians(deltaLon / 2)), 2));
198
199                double c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
200                distance = radius * c;
201
202                return distance;
203        }
204
205        /**
206         * 
207         * @param deltaTime
208         *            The between two data points
209         * @return The average speed between two points
210         */
211        private double calculateSpeed(long deltaTime) {
212                double speed;
213                if (deltaTime == 0) {
214                        speed = 0;
215                } else {
216                        speed = (distance / deltaTime);
217                }
218                return speed;
219        }
