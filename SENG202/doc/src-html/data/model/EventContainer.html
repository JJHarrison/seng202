001package data.model;
002
003import java.text.SimpleDateFormat;
004import java.util.Date;
005import java.util.HashMap;
006import java.util.LinkedList;
007
008/**
009 * This class is abstract version of an activity event container. The activity
010 * event container consists of activity days containing Event objects
011 * representing activity events imported from a fitness tracking device.
012 * 
013 * @author Fitr.Team
014 */
015public class EventContainer {
016        private HashMap<String, LinkedList<Event>> days;
017
018        /**
019         * Constructs a hash map to store the activity events for each day in.
020         */
021        public EventContainer() {
022                days = new HashMap<String, LinkedList<Event>>();
023        }
024
025        /**
026         * Add a activity event to the event container.
027         * 
028         * @param e
029         *            The activity event to be added to the container.
030         */
031        public void addEvent(Event e) {
032                String dateString = dateString(e.getStartTime().getTime());
033                if (days.containsKey(dateString)) {
034                        // find the correct position to add the event
035                        LinkedList<Event> day = days.get(dateString);
036                        for (int i = 0; i < days.size(); i++) {
037                                // reached the right index.
038                                if (e.getStartTime().before(day.get(i).getStartTime())) {
039                                        // add e at i
040                                        day.add(i, e);
041                                }
042                                // e is the latest event
043                                else if (i == days.size() - 1) {
044                                        // add e to the end of the list
045                                        day.add(e);
046                                }
047                        }
048                } else {
049                        LinkedList<Event> day = new LinkedList<Event>();
050                        day.add(e);
051                        days.put(dateString, day);
052                }
053        }
054
055        /**
056         * Returns the linked list that contains the activity events for the day.
057         * 
058         * @param date
059         *            The day that the caller wants the activity events for.
060         * @return The linked list containing activity events
061         */
062        public LinkedList<Event> getEvents(Date date) {
063                String dateString = dateString(date);
064                if (days.containsKey(dateString)) {
065                        return days.get(dateString);
066                } else {
067                        return null;
068                }
069        }
070
071        /**
072         * Returns a date string in the format dd/mm/yyyy to be used as the key for
073         * the hash map.
074         * 
075         * @param date
076         *            The date object to be formatted
077         * @return The formatted date string
078         */
079        private String dateString(Date date) {
080                return new SimpleDateFormat("dd/MM/yyyy").format(date.getTime());
081        }
082}
