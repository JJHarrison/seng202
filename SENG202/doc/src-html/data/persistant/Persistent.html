021        private static Preferences prefs = Preferences.userRoot().node("/Fitr");
022        private static ObservableList<User> users = FXCollections.observableList(new ArrayList<User>());
023        private static ObservableList<String> userNames = FXCollections.observableList(new ArrayList<String>());
024
025        /**
026         * sets the FilePath preference to a Fitr directory at location of filePath
027         * @param filePath
028         */
029        public static void setFilePath(String filePath) {
030                // filePath will be obtained from a file chooser so it will always be an existing file path so doesn't need to be checked
031                prefs.put("FilePath", filePath + "/Fitr/");
032                try {
033                        prefs.flush();
034                } catch (BackingStoreException e) {
035                        e.printStackTrace();
036                }
037                
038                
039                //only creates new directory if the selected one doesn't already exist
040                if(!new File(getFilePath()).exists()) {
041                        setupDirectory();
042                } else {
043                        //if the Fitr dir already exists then use it.
044                        Persistent.init();
045                }
046        }
047        
048        /**
049         * sets up a new Fitr directory with a users sub directory at the file path 
050         */
051        public static void setupDirectory() {
052                if(getFilePath() != null) {
053                        new File(getFilePath()).mkdirs();
054                }
055        }
056        
057        /** 
058         * gets the file path from preferences 
059         * @return FilePath 
060         */
061        public static String getFilePath() {
062                return prefs.get("FilePath", null);
063        }
064        
065        /**
066         * returns the file path to the current users profile
067         * @return profileFilePath
068         */
069        public static String getProfileFilePath(String userName) {
070                return getFilePath() + userName + "/" + userName + ".fitr";
071        }
072        
073        /**
074         * returns the file path to the current users activity data
075         * @return ActivityFilePath
076         */
077        public static String getActivityFilePath(String userName) {
078                return getFilePath() + userName + "/" + userName + "Activity.fitr";
079        }
080        
081        /**
082         * checks if a valid file path has been set
083         * @return pathSet
084         */
085        public static boolean filePathSet() {
086                boolean pathSet = true;
087                File filePath;
088                if (getFilePath() == null) {
089                        filePath = null;
090                } else {
091                        filePath = new File(getFilePath());
092                }
093                                
094                if(filePath == null || !filePath.exists()) {
095                        pathSet = false;
096                }
097                return pathSet;
098        }
099        
100        
101        /**
102         * creates a new directory in users with name user to store information in 
103         * also adds a user.json and activity.json file to save data to
104         * @param userName
105         */
106        public static void newUser(User user) throws Exception {
107                String userName = user.getName();
108
109                if (! userNames.contains(userName)) {
110                        new File(getFilePath() + userName).mkdir();
111                        
112                        try {
113                                new File(getProfileFilePath(userName)).createNewFile();
114                                // should go in its own try catch block?
115                                new File(getActivityFilePath(userName)).createNewFile();
116                        } catch (IOException e) {
117                                e.printStackTrace();    
118                        }
119                        
120                        users.add(user);
121                        userNames.add(user.getName());
122                        Saver.SaveUser(user);
123                        
124                } else {
125                        System.out.println("User has already beeen created. try differnt username");
126                        //throw new Exception("User already exists");
127                }
128                
129                
130        }

156        /**
157         * gets the user names of all the profiles
158         * @return userNames
159         */
160        public static ObservableList<String> getUserNames() {
161                for(User user : users) {
162                        userNames.add(user.getName());  
163                }
164                return userNames;
165        }

167        /**
168         * loads all the profiles from the users directory into the ObservableList<User> users
169         */
170        public static void init() {
171                if(getFilePath() != null && new File(getFilePath()).exists()) {
172                        System.out.println("initialising");
173                        File filePath = new File(getFilePath());
174                        
175                        //get files / directory in Fitr directory (gets users)
176                        File[] Files = filePath.listFiles();
177
178                        for(File file : Files) {
179                                System.out.println(file);
180                                String userName = file.getName();
181                                
182                                if(new File(file + "/" + userName + ".fitr").exists()) { //check if the file is a user
183                                        User newUser = Loader.loadUserProfile(new File(file + "/" + userName + ".fitr"));
184                                        users.add(newUser);
185                                        userNames.add(newUser.getName());
186                                }
187                        }
188                }
189        }
