001package dataImport;
002
003import java.io.BufferedReader;
004import java.io.File;
005import java.io.FileInputStream;
006import java.io.FileNotFoundException;
007import java.io.IOException;
008import java.io.InputStream;
009import java.io.InputStreamReader;
010import java.util.ArrayList;
011import java.util.Calendar;
012import java.util.GregorianCalendar;
013
014import dataModel.DataPoint;
015import dataModel.Event;
016
017/**
018 * This class provides a way of reading activity events provided by a fitness
019 * tracking device. The activity data needs to be in a csv format consisting of
020 * date, time, heart rate, latitude, longitude & altitude.
021 * 
022 * @author Fitr.Team
023 */
024public class FileLoader {
025        private InputStream inputStream;
026        
027        public void clearStream(){
028                inputStream=null;
029        }
030        
031        /**
032         * Gets the input file
033         * @return
034         */
035        public InputStream getStream() {
036                return inputStream;
037        }
038
039        public FileLoader(File file) {
040                try {
041                        inputStream = new FileInputStream(file);
042                } catch (FileNotFoundException e) {
043                        e.printStackTrace();
044                        clearStream();
045                } catch (NullPointerException e) {
046                        // TODO: handle exception
047                }
048        }
049
050        public FileLoader() {
051                inputStream = this.getClass().getResourceAsStream(
052                                "seng202_2014_example_data.csv");
053        }
054
055        private ArrayList<Event> events = new ArrayList<Event>();
056
057        /**
058         * reads a csv file and create a new event at each #start add all following
059         * data points to the event
060         */
061        public void load() {
062                BufferedReader br = null;
063                String line = "";
064                DataPoint lastPoint = null;
065                Event currentEvent = new Event("");
066
067                try {
068                        br = new BufferedReader(new InputStreamReader(inputStream));
069
070                        while ((line = br.readLine()) != null) {
071                                String[] dataLine = line.split(",");
072                                if (!line.isEmpty()) {
073                                        if (dataLine[0].contains("#start")) {
074                                                currentEvent = new Event(dataLine[1]);
075                                                events.add(currentEvent);
076                                                lastPoint = null;
077                                        } else {
078                                                String[] dateString = dataLine[0].split("/");
079                                                String[] time = dataLine[1].split(":");
080        
081                                                // months start from 0...
082                                                Calendar date = new GregorianCalendar(
083                                                                Integer.parseInt(dateString[2]), // Year
084                                                                Integer.parseInt(dateString[1]), // Month
085                                                                Integer.parseInt(dateString[0]), // Day
086                                                                Integer.parseInt(time[0]), // Hour
087                                                                Integer.parseInt(time[1]), // Minute
088                                                                Integer.parseInt(time[2])); // Second
089        
090                                                int heartrate = Integer.parseInt(dataLine[2]);
091                                                double latitude = Double.parseDouble(dataLine[3]);
092                                                double longitude = Double.parseDouble(dataLine[4]);
093                                                double altitude = Double.parseDouble(dataLine[5]);
094        
095                                                DataPoint point = new DataPoint(date, heartrate, latitude,
096                                                                longitude, altitude, lastPoint);
097                                                currentEvent.addDataPoint(point);
098                                                lastPoint = point;
099                                        }
100                                }
101                        }
102
103                } catch (FileNotFoundException e) {
104                        e.printStackTrace();
105                        System.out.println("File Not Found");
106                } catch (IOException e) {
107                        e.printStackTrace();
108                        System.out.println("couldnt read line");
109                }
110
111        }
112
113        public ArrayList<Event> getEvents() {
114                return events;
115        }
116}
