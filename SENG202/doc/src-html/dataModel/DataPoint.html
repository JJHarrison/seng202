012/**
013 * This class provides an abstract version of points provided by a fitness tracking device. Each dataPoint 
014 * consists of attributes related to the actual values in the real world at that time point. 
015 * @author Fitr.Team
016 */
017public class DataPoint {
018        private Calendar date;
019        private int heartRate;
020        private double latitude;
021        private double longitude;
022        private double altitude;
023        private double speed;
024        private double distance;
025        
026        /**
027         * Constructor.
028         * @param date The current date at this point.
029         * @param heartrate The current heart rate at this point.
030         * @param latitude The current latitude at this point.
031         * @param longitude The current longitude at this point.
032         * @param altitude The current altitude at this point.
033         * @param previousPointPoint The previous point, used for distance calculations.
034         */
035        public DataPoint(Calendar date, int heartrate, double latitude, double longitude, double altitude, DataPoint previousPointPoint) {
036                this.date = date;
037                this.heartRate = heartrate;
038                this.latitude =latitude;
039                this.longitude = longitude;
040                this.altitude = altitude;
041                
042                if (previousPointPoint != null) {
043                        this.distance = calculateDistance(previousPointPoint);
044                        this.speed = calculateSpeed(calculateDeltaTime(previousPointPoint));
045                } else {
046                        this.distance = 0.0;
047                        this.speed = 0.0;
048                }
049        }
050
051        /**
052         * Calculates the change in time between two points.
053         * @param previousPoint The point previous to this point in an event.
054         * @return The change in time (seconds).
055         */
056        private long calculateDeltaTime(DataPoint previousPoint) {
057                long previousTime = previousPoint.getDate().getTimeInMillis();
058                return ((date.getTimeInMillis() - previousTime) / 1000);
059        }
060        
061        /**
062         * Calculates the distance from the current point to the previousPoint in meters.
063         * @param previousPoint The point from which to calculate the distance from 
064         * @return The distance in meters from the previous point
065         */
066        private double calculateDistance(DataPoint previousPoint) {
067                double distance = 0;
068                double radius = 6373 * 1000; // Converted to meters
069                double latPrev = previousPoint.getLatitude();
070                double lonPrev = previousPoint.getLongitude();
071                
072                double deltaLat = latPrev - latitude;
073                double deltaLon = lonPrev - longitude;
074                
075                double a = Math.pow(Math.sin(Math.toRadians(deltaLat / 2)), 2) + (Math.cos(Math.toRadians(latitude))
076                                * Math.cos(Math.toRadians(latPrev)) * Math.pow(Math.sin(Math.toRadians(deltaLon / 2)),2));
077                
078                double c = 2 * Math.atan2( Math.sqrt(a), Math.sqrt(1-a) );
079                distance = radius * c;
080                
081                return distance;
082        }
083        
084        /**
085         * 
086         * @param deltaTime The between two data points
087         * @return The average speed between two points
088         */
089        private double calculateSpeed(long deltaTime) {
090                double speed;
091                if(deltaTime == 0) {
092                        speed = 0;
093                } else {
094                        speed =  (distance / deltaTime);
095                }
096                return speed;
097        }
098
099        /**
100         * Gets the date at a particular point.
101         * @return the date
102         */
103        public Calendar getDate() {
104                return date;
105        }
106
107        /**
108         * Gets the heart rate at a particular point.
109         * @return the heartRate
110         */
111        public int getHeartRate() {
112                return heartRate;
113        }
114
115        /**
116         * Gets the latitude at a particular point.
117         * @return the latitude
118         */
119        public double getLatitude() {
120                return latitude;
121        }
122
123        /**
124         * Gets the longitude at a particular point.
125         * @return the longitude
126         */
127        public double getLongitude() {
128                return longitude;
129        }
130
131        /**
132         * Gets the altitude at a particular point.
133         * @return the altitude
134         */
135        public double getAltitude() {
136                return altitude;
137        }
138        
139        /**
140         * Gets the average speed (m/s) from the previous point to this point.
141         * @return the speed
142         */
143        public double getSpeed() {
144                return speed;
145        }
146
147        /**
148         * Gets the distance (m) from the previous point to this point.
149         * @return the distance
150         */
151        public double getDistance() {
152                return distance;
153        }
154        
155        /**
156         * Returns a string representation of the timestamp for the data point as "hh:mm:ss" 
157         * @return time of the data point
158         */
159        public String getTimeString(){
160                SimpleDateFormat tf = new SimpleDateFormat("HH:mm:ss");
161                return(tf.format(date.getTime()));
162                
163        }
164        
165        /**
166         * Returns a string representation of the date for the data point as "dd/mm/yyyy"
167         * @return date of the data point 
168         */
169        public String getDateString(){
170                //The calendar takes january being the 0th month and december the 11th
171                SimpleDateFormat df = new SimpleDateFormat("dd/MM/yyyy");
172                return(df.format(date.getTime()));
173        }
174

177                return new SimpleStringProperty(getDateString());
178        }

182                return new SimpleStringProperty(getTimeString());

202                return new SimpleDoubleProperty(getSpeed());
203        }
204

206        public Property<?> getHeartRateProperty() {
207                return new SimpleIntegerProperty(getHeartRate());
208        }
