001package dataModel;
002
003import java.text.SimpleDateFormat;
004import java.util.ArrayList;
005import java.util.Calendar;
006
007import dataModel.DataPoint;
008
009/**
010 * This class provides an abstract version of activity events provided by a fitness tracking device. 
011 * Each event consists of a collection of dataPoints and attributes that relate to the activity event.  
012 * @author Fitr.Team
013 */
014public class Event {
015        private String eventName;
016        private Calendar startTime;
017        private Calendar finishTime;
018        private int numPoints;
019        private double distance;
020        private double maxSpeed;
021        private double averageSpeed;
022        private int totalHeartRate;
023        private ArrayList<DataPoint> points = new ArrayList<DataPoint>();
024        
025        /**
026         * Constructor.
027         * @param eventName The name of the event.
028         */
029        public Event(String eventName) {
030                this.eventName = eventName;
031        }
032        
033        /**
034         * Prints the name of the activity event.
035         */
036        public void printEventName(){
037                System.out.println(eventName);
038        }
039        
040        /**
041         * Sets the start time of the activity event.
042         * @param c The start time of the event.
043         */
044        public void setStartTime(Calendar c) {
045                this.startTime = c;
046        }
047        
048        /**
049         * Sets the finish time of the activity event.
050         * @param c The finish time of the event.
051         */
052        public void setFinishTime(Calendar c) {
053                this.finishTime = c;
054        }
055        
056        /**
057         * Adds a dataPoint to the activity event.
058         * @param p The dataPoint to be added to the activity event.
059         */
060        public void addDataPoint(DataPoint p) {
061                points.add(p);
062                distance += p.getDistance();
063                averageSpeed  = calculateAverageSpeed(p);
064                numPoints += 1;
065                finishTime = p.getDate();
066                startTime = this.getDataPoints().get(0).getDate();
067                totalHeartRate += p.getHeartRate();
068                
069                if(maxSpeed < p.getSpeed()) {
070                        maxSpeed = p.getSpeed();
071                }
072        }
073        
074        /**
075         * Gets a summary of the activity event consisting of the event name, date, start time, end time, average speed,
076         * calories burned, average heart rate. The summary is returned as a String.
077         * @return A String summary of the activity event. 
078         */
079        public String getSummary() {
080                SimpleDateFormat tf = new SimpleDateFormat("HH:mm:ss");
081                SimpleDateFormat df = new SimpleDateFormat("dd/MM/yyyy");
082                
083                String startTime = tf.format(this.startTime.getTime());
084                String endTime = tf.format(this.finishTime.getTime());
085                String date = df.format(this.startTime.getTime());
086                
087                String summary = String.format("Event Name: %s\nDate: %s\nStart Time: %s\nEnd Time: %s\nAverageSpeed: %.2f m/s\nMax Speed: %.2f m/s\nDistance:"
088                                + " %.0f m\nCalories Burned: %.0f\nAverage HR: %d", eventName, date, startTime, endTime, getAverageSpeed(), maxSpeed, distance, getCaloriesBurned(), getAverageHeartRate());
089                
090                return summary;
091        }
092        
093        private double calculateAverageSpeed(DataPoint p) {
094                double newAverageSpeed = ((numPoints * averageSpeed) + p.getSpeed()) / (numPoints + 1);
095                return newAverageSpeed;
096        }
097        
098        /**
099         * Returns the average speed of the activity event.
100         * @return The average speed in meters per second
101         */
102        public double getAverageSpeed() {
103                return averageSpeed;
104        }
105        
106        
107        /**
108         * Returns the average heart rate of the activity event.
109         * @return The average heart rate in beats per minute.
110         */
111        public int getAverageHeartRate() {
112                return totalHeartRate / numPoints;
113        }
114        
115        /**
116         * Returns the max speed of the activity event.
117         * @return The max speed in meters per second.
118         */
119        public double getMaxSpeed() {
120                return maxSpeed;
121        }
122        
123        /**
124         * Returns the total distance traveled in the activity event.
125         * @return The total distance in meters.
126         */
127        public double getDistance() {
128                return distance;
129        }
130        
131        /**
132         * Returns the name of the activity event.
133         * @return A String of containing the activity event name.
134         */
135        public String getEventName() {
136                return this.eventName;
137        }
138        
139        /**
140         * Returns the start time of the activity event.
141         * @return The start time of the activity event as a calendar object.
142         */
143        public Calendar getStartTime() {
144                return this.startTime;
145        }
146        
147        /**
148         * Returns the finish time of the activity event.
149         * @return The finish time of the activity event as a calendar object 
150         */
151        public Calendar getFinishTime() {
152                return this.finishTime;
153        }
154        
155        /**
156         * Returns the duration of the activity event.
157         * @return The duration for the activity event in hours.
158         */
159        public double getDuration() {
160                return (finishTime.getTimeInMillis() - startTime.getTimeInMillis()) / (1000.0 * 60 * 60);
161        }
162        
163        /**
164         * Returns the data points of the activity event.
165         * @return The dataPoints inside the activity event.
166         */
167        public ArrayList<DataPoint> getDataPoints() {
168                return this.points;
169        }
170        
171        /**
172         * Returns an estimation of calories burned during the activity event.
173         * @return The calories burned for the activity events.
174         */
175        public double getCaloriesBurned() {
176                int weight = 75;
177                double runMET = 7.5;
178                double timeInHours = getDuration();
179                
180                double calories = weight * runMET * timeInHours;
181                return calories;
182        }
183        
184        /**
185         * Returns a String representation of the events name.
186         * @return The events name as a String.
187         */
188        @Override
189        public String toString() {
190                return getEventName() + "\n" + new SimpleDateFormat("dd/MM/yyyy").format(getStartTime().getTime());
191        }
192}
