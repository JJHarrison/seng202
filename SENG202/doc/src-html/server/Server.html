001package server;
002
003import java.io.EOFException;
004import java.io.IOException;
005import java.io.ObjectInputStream;
006import java.io.ObjectOutputStream;
007import java.net.InetAddress;
008import java.net.ServerSocket;
009import java.net.Socket;
010import java.text.SimpleDateFormat;
011import java.util.Calendar;
012
013public class Server {
014        private ServerSocket server;
015        private Socket connection;
016        private ObjectInputStream input;
017        private ObjectOutputStream output;
018        // private String host = "localhost";
019        final private int portNumber = 8888;
020        final private int maxQueue = 10;
021
022        /*
023         * Still need to look into threading for the client sockets
024         */
025
026        /**
027         * Starts a server that will endlessly wait for connections
028         */
029        public void startServer() {
030                try {
031                        server = new ServerSocket(portNumber, maxQueue);
032                        while (true) {
033                                try {
034                                        waitForConnection();
035                                        setupStreams();
036                                        whileTransfering();
037                                        System.out.println(startMessage() + " Connection ended!");
038                                        System.out.println("-----------------------------------"
039                                                        + "--------------------");
040                                } catch (EOFException e) {
041                                        System.out.println(startMessage() + " Connection ended!");
042                                        System.out.println("-----------------------------------"
043                                                        + "--------------------");
044                                } finally {
045                                        closeStuff();
046                                }
047                        }
048                } catch (IOException e) {
049                        e.printStackTrace();
050                }
051        }
052
053        /**
054         * Waits for a connection to be formed with the client.
055         * 
056         * @throws IOException
057         */
058        public void waitForConnection() throws IOException {
059                System.out.println(startMessage()
060                                + " Waiting for a client to connect to ["
061                                + InetAddress.getLocalHost().getCanonicalHostName() + "]...");
062                // Once a client connects a socket is open for the server and client
063                connection = server.accept();
064                System.out.println(String.format(startMessage()
065                                + " Now connected to %s", connection.getInetAddress()
066                                .getHostName()));
067        }
068
069        /**
070         * Sets up the output/input streams to send/receive objects with the client.
071         * 
072         * @throws IOException
073         */
074        public void setupStreams() throws IOException {
075                output = new ObjectOutputStream(connection.getOutputStream());
076                output.flush();
077                input = new ObjectInputStream(connection.getInputStream());
078                System.out
079                                .println(startMessage() + " Streams are now ready to be used");
080        }
081
082        /*
083         * This will need to be changed once we are sending actual information that
084         * the server will store
085         */
086        public void whileTransfering() throws IOException {
087                String clientMessage = null;
088                System.out.println(startMessage() + " Ready for transfer...\n");
089                do {
090                        try {
091                                // Will need to send a string to from the client describing what
092                                // is being sent
093                                // maybe a comment that would also be stored with the file
094                                clientMessage = (String) input.readObject();
095                                // Prints that we have received the object
096                                System.out.println(startMessage() + " User sent: "
097                                                + clientMessage);
098                                /*
099                                 * This is where we need to process the file being sent from the
100                                 * client... --We must decide where it will be stored --We must
101                                 * decide how it will be stored --Probably need to have some
102                                 * container/storage class --I think we will also need some
103                                 * checks here, or decide if that client will perform a check
104                                 * also/instead
105                                 */
106                                // send a confirmation to the client that the message was
107                                // received correctly
108                                sendMessage(clientMessage);
109                        } catch (ClassNotFoundException e) {
110                                System.out.println(startMessage()
111                                                + " I don't know what to do with that type of object");
112                        }
113                } while (!clientMessage.equals("END"));
114        }
115
116        /**
117         * Closes the streams and socket once the connection is lost
118         */
119        public void closeStuff() {
120                try {
121                        input.close();
122                        output.close();
123                        connection.close();
124                } catch (IOException e) {
125                        e.printStackTrace();
126                }
127        }
128
129        /**
130         * Sends a confirmation message to the client that their message was
131         * received.
132         * 
133         * @param message
134         *            The message that was received from the client.
135         */
136        private void sendMessage(String message) {
137                try {
138                        output.writeObject(message);
139                        output.flush();
140                        System.out.println(startMessage()
141                                        + " Sent confirmation to the <Client>\n");
142                } catch (IOException e) {
143                        System.out.println(startMessage()
144                                        + " No confirmation was sent to the <Client>");
145                }
146        }
147
148        /**
149         * Gets the current time on the server.
150         * 
151         * @return
152         */
153        public String getCurrentTime() {
154                SimpleDateFormat df = new SimpleDateFormat("HH:mm:ss");
155                Calendar cal = Calendar.getInstance();
156                return df.format(cal.getTime());
157        }
158
159        /**
160         * Creates a message consisting of [currentTime]<Server> for the server
161         * output messages.
162         * 
163         * @return The opening string for server output messages.
164         */
165        public String startMessage() {
166                return "[" + getCurrentTime() + "]<Server>";
167        }
168
169        public static void main(String[] args) {
170                Server s = new Server();
171                s.startServer();
172
173        }
174}
