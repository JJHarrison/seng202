003import java.util.ArrayList;

006
007import dataModel.Event;

010
011/**
012 * Tests for the event class functionality
013 * @author James
014 *
015 */
016public class EventTest extends TestCase {
017        private Event e;
018        private ArrayList<DataPoint> points;
019        private Calendar c1;
020        private Calendar c2;
021        private DataPoint p1;
022    private DataPoint p2;
023        
024        /**
025         * Sets up the events to be tested
026         */
027        @Override
028    protected void setUp() throws Exception {
029        super.setUp();
030        
031        e = new Event("My Event");
032        
033        // set start and finish times 3 minutes apart
034                c1 = new GregorianCalendar(
035                                2005, // Year 
036                                5, // Month
037                                10, // Day
038                                23, // Hour
039                                42, // Minute
040                                28); // Second
041                c2 = new GregorianCalendar(
042                                2005, // Year 
043                                5, // Month
044                                10, // Day
045                                23, // Hour
046                                45, // Minute
047                                28); // Second
048                
049        e.setStartTime(c1);
050        e.setFinishTime(c2);
051        
052        // set up data points
053        points = new ArrayList<DataPoint>();
054                Calendar c3 = new GregorianCalendar(
055                                2005, // Year 
056                                5, // Month
057                                10, // Day
058                                23, // Hour
059                                42, // Minute
060                                28); // Second
061                Calendar c4 = new GregorianCalendar(
062                                2005, // Year 
063                                5, // Month
064                                10, // Day
065                                23, // Hour
066                                43, // Minute
067                                5); // Second
068                
069                p1 = new DataPoint(c3, 120, 30.2553368, -97.83891084, 50.0, null);
070                p2 = new DataPoint(c4, 125, 30.25499189, -97.83913958, 51.0, p1);
071        points.add(p1);
072        points.add(p2);
073        e.addDataPoint(p1);
074        e.addDataPoint(p2);
075        
076    }
077        
078        /**
079         * Tests the getEventName function to make sure it returns the correct name of the event
080         */
081        public void testGetEventName() {
082                assertEquals(e.getEventName(), "My Event");
083        }
084        
085        /**
086         * Tests the getDataPoints function to make sure it returns the dataPoints of the event. 
087         */
088        public void testGetDataPoints() {
089                assertEquals(e.getDataPoints(), points);
090        }
091        
092        
093        //run this last or you will cause the other test to fail
094        /**
095         * Tests the addDataPoint function.
096         */
097        public void testAddDataPoint() {
098                assertEquals(p2, points.get(points.size()-1));
099                points.add(p1);
100                assertEquals(p1, points.get(points.size()-1));
101        }
102}
