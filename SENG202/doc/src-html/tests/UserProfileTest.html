001package tests;
002
003import java.util.GregorianCalendar;
004
005import junit.framework.TestCase;
006import userModel.UserProfile;
007import userModel.UserProfile.Gender;
008
009/**
010 * Tests the functionality of the UserProfile class
011 * 
012 * @author Fitr.Team
013 */
014public class UserProfileTest extends TestCase {
015
016        private UserProfile john;
017
018        /**
019         * Sets up the profile to be tested.
020         */
021        protected void setUp() throws Exception {
022                super.setUp();
023
024                // John Key's actual birthday
025                GregorianCalendar dob = new GregorianCalendar(1961, 8, 9);
026                john = new UserProfile("John Key", dob, Gender.MALE);
027                john.setWeight(81.2);
028                john.setHeight(1.8);
029
030        }
031
032        /**
033         * Tests the getAge function to make sure it returns the right age
034         */
035        public void testGetAge() {
036                assertEquals(john.getAge(), 53);
037        }
038
039        /**
040         * Tests the BMI calculation
041         */
042        public void testGetBMI() {
043                double bmi = john.getBMI();
044                assertTrue(bmi > 24.9);
045                assertTrue(bmi < 25.1);
046        }
047
048        /**
049         * Tests if changing the weight automatically updates the BMI of the user
050         */
051        public void testChangeWeight() {
052                john.setWeight(162.4);
053                double bmi = john.getBMI();
054                assertTrue(bmi > 49.8);
055                assertTrue(bmi < 50.2);
056        }
057
058        /**
059         * Tests if changing the height automatically updates the BMI of the user
060         */
061        public void testChangeHeight() {
062                john.setHeight(2.0);
063                double bmi = john.getBMI();
064                assertTrue(bmi > 20);
065                assertTrue(bmi < 21);
066        }
067
068}
