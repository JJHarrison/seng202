001package view;
002
003import java.io.File;
004import java.util.ArrayList;
005
006import jfx.messagebox.MessageBox;
007import dataImport.FileLoader;
008import dataModel.DataPoint;
009import dataModel.Event;
010import javafx.beans.value.ChangeListener;
011import javafx.beans.value.ObservableValue;
012import javafx.collections.FXCollections;
013import javafx.collections.ObservableList;
014import javafx.event.ActionEvent;
015import javafx.event.EventHandler;
016import javafx.fxml.FXML;
017import javafx.scene.control.ListView;
018import javafx.scene.control.MenuItem;
019import javafx.scene.control.TableColumn;
020import javafx.scene.control.TableView;
021import javafx.scene.control.TextArea;
022import javafx.scene.control.cell.PropertyValueFactory;
023import javafx.stage.FileChooser;
024import javafx.stage.FileChooser.ExtensionFilter;
025import javafx.stage.Stage;
026
027/**
028 * The current controller for the Fitr application.
029 * 
030 * @author Daniel van Wichen
031 *
032 */
033public class FitrController {
034        @FXML
035        private Stage stage;
036
037        @FXML
038        private TableView<DataPoint> tableView;
039
040        @FXML
041        private TableColumn<DataPoint, String> colDate;
042        @FXML
043        private TableColumn<DataPoint, String> colTime;
044        @FXML
045        private TableColumn<DataPoint, Double> colLatitude;
046        @FXML
047        private TableColumn<DataPoint, Double> colLongitude;
048        @FXML
049        private TableColumn<DataPoint, String> colDistance;
050        @FXML
051        private TableColumn<DataPoint, String> colSpeed;
052        @FXML
053        private TableColumn<DataPoint, Integer> colHR;
054
055        @FXML
056        private ListView<Event> eventSelector;
057
058        @FXML
059        private TextArea eventSummary;
060
061        @FXML
062        private MenuItem menuImport;
063        @FXML
064        private MenuItem menuClear;
065        @FXML
066        private MenuItem menuClose;
067        @FXML
068        private MenuItem menuAbout;
069
070        public FitrController() {
071        }
072
073        @FXML
074        private void initialize() {
075                menuImport.setOnAction(new EventHandler<ActionEvent>() {
076
077                        public void handle(ActionEvent event) {
078                                eventSelector.getSelectionModel().clearAndSelect(0);
079                                File file = importFile();
080                                if (file != null) {
081                                        loadFile(file);
082                                }
083                        }
084                });
085
086                menuClear.setOnAction(new EventHandler<ActionEvent>() {
087
088                        public void handle(ActionEvent event) {
089                                clear();
090
091                        }
092                });
093
094                menuClose.setOnAction(new EventHandler<ActionEvent>() {
095
096                        public void handle(ActionEvent event) {
097                                System.exit(0);
098
099                        }
100                });
101
102                menuAbout.setOnAction(new EventHandler<ActionEvent>() {
103
104                        public void handle(ActionEvent event) {
105
106                                MessageBox.show(stage, "Version 1.0\n"
107                                                + "\nThis is a prototype\n" + "USE WITH CARE!",
108                                                "About Fitr", MessageBox.OK );
109                        }
110                });
111
112                FileLoader fLoader = new FileLoader();
113                fLoader.load();
114
115                ObservableList<Event> events = FXCollections.observableArrayList();
116                events.setAll(fLoader.getEvents());
117                eventSelector.setItems(events);
118
119                eventSelector.getSelectionModel().selectedItemProperty()
120                                .addListener(new ChangeListener<Event>() {
121
122                                        public void changed(
123                                                        ObservableValue<? extends Event> observable,
124                                                        Event oldValue, Event newValue) {
125                                                if (newValue != null) {
126                                                        eventSummary.setText(newValue.getSummary());
127                                                        ObservableList<DataPoint> data = FXCollections
128                                                                        .observableArrayList(newValue
129                                                                                        .getDataPoints());
130                                                        tableView.itemsProperty().setValue(data);
131                                                } else {
132                                                        DataPoint nullPoint = null;
133                                                        eventSummary.setText("");
134                                                        tableView.itemsProperty().setValue(FXCollections.observableArrayList(nullPoint));
135                                                }
136                                        }
137                                });
138
139                eventSelector.getSelectionModel().clearAndSelect(0);
140
141                colDate.setCellValueFactory(new PropertyValueFactory<DataPoint, String>("getDate"));
142                colTime.setCellValueFactory(new PropertyValueFactory<DataPoint, String>("getTime"));
143                colLatitude.setCellValueFactory(new PropertyValueFactory<DataPoint, Double>("getLatitude"));
144                colLongitude.setCellValueFactory(new PropertyValueFactory<DataPoint, Double>("getLongitude"));
145                colDistance.setCellValueFactory(new PropertyValueFactory<DataPoint, String>("getDistance"));
146                colSpeed.setCellValueFactory(new PropertyValueFactory<DataPoint, String>("getSpeed"));
147                colHR.setCellValueFactory(new PropertyValueFactory<DataPoint, Integer>("getHeartRate"));
148        }
149
150        private File importFile() {
151                FileChooser fileChooser = new FileChooser();
152                fileChooser.setTitle("Open Activity File");
153                ArrayList<ExtensionFilter> filter = new ArrayList<ExtensionFilter>();
154                filter.add(new ExtensionFilter("CSV file", "*.csv"));
155                fileChooser.getExtensionFilters().setAll(FXCollections.observableList(filter));
156                File file = fileChooser.showOpenDialog(new Stage());
157                return file;
158        }
159
160        private void loadFile(File file) {
161                FileLoader fLoader = new FileLoader(file);
162                if (fLoader.getStream() != null) {
163                        fLoader.load();
164                        ObservableList<Event> events = FXCollections.observableArrayList();
165                        events.setAll(fLoader.getEvents());
166                        eventSelector.setItems(events);
167                        eventSelector.getSelectionModel().clearAndSelect(0);
168                }
169        }
170
171        private void clear() {
172                eventSelector.setItems(null);
173        }
174
175}
