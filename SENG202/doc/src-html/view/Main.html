006package view;
007
008
009
010import java.io.IOException;
011
012import javafx.application.Application;
013import javafx.fxml.FXMLLoader;
014import javafx.scene.Scene;
015import javafx.scene.image.Image;
016import javafx.scene.layout.BorderPane;
017import javafx.stage.Stage;
018
019public class Main extends Application {
020
021        private Stage stage;
022        private BorderPane rootLayout;
023
024        @Override
025        public void start(Stage primaryStage) {
026
027                stage = primaryStage;
028                stage.setTitle("Fitr");
029                
030
031                // Minimum size of the stage.
032                stage.setMinHeight(768);
033                stage.setMinWidth(1366);
034                
035                try {
036                        this.stage.getIcons().add(
037                                        new Image("/resources/heart11.png"));
038                } catch (Exception e) {
039                        e.printStackTrace();
040                }
041
042                initRootLayout();
043        }
044
045        /**
046         * Initializes the root layout.
047         */
048        public void initRootLayout() {
049                try {   
050                        // Load root layout from FXML file.
051                        FXMLLoader loader = new FXMLLoader();
052                        loader.setLocation(Main.class.getResource("Main.fxml"));
053                        rootLayout = (BorderPane) loader.load();
054
055                        // Show the scene containing the root layout.
056                        Scene scene = new Scene(rootLayout);
057                        stage.setScene(scene);
058                        stage.centerOnScreen();
059                        stage.show();
060
061                } catch (IOException e) {
062                        e.printStackTrace();
063                }
064        }
065
066        public static void main(String[] args) {
067                launch(args);
068        }
069}
