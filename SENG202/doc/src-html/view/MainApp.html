006package view;
007
008import java.io.IOException;
009
010import javafx.application.Application;
011import javafx.fxml.FXMLLoader;
012import javafx.scene.Scene;
013import javafx.scene.image.Image;
014import javafx.scene.layout.BorderPane;
015import javafx.stage.Stage;
016
017public class MainApp extends Application {
018
019        private Stage primaryStage;
020        private BorderPane rootLayout;
021
022        @Override
023        public void start(Stage primaryStage) {
024                this.primaryStage = primaryStage;
025                this.primaryStage.setTitle("Fitr.aw");
026                try {
027                        this.primaryStage.getIcons().add(
028                                        new Image("/resources/heart11.png"));
029                } catch (Exception e) {
030                        e.printStackTrace();
031                }
032
033                initRootLayout();
034        }
035
036        /**
037         * Initializes the root layout.
038         */
039        public void initRootLayout() {
040                try {
041                        // Load root layout from fxml file.
042                        FXMLLoader loader = new FXMLLoader();
043                        loader.setLocation(MainApp.class.getResource("FitrView.fxml"));
044                        rootLayout = (BorderPane) loader.load();
045
046                        // Show the scene containing the root layout.
047                        Scene scene = new Scene(rootLayout);
048                        primaryStage.setScene(scene);
049                        primaryStage.show();
050
051                } catch (IOException e) {
052                        e.printStackTrace();
053                }
054        }
055
056        public static void main(String[] args) {
057                launch(args);
058        }
059}
