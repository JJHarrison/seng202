001package view.tile;
002
003import javafx.event.ActionEvent;
004import javafx.fxml.FXML;
005import javafx.scene.Node;
006import javafx.scene.control.ToggleButton;
007import javafx.scene.control.ToggleGroup;
008import javafx.scene.input.MouseEvent;
009import javafx.scene.layout.AnchorPane;
010import javafx.scene.layout.StackPane;
011import javafx.scene.layout.VBox;
012
013public class TileController {
014
015    /** Holder of a switchable view. */
016    @FXML
017    StackPane tileView;
018    @FXML
019    VBox tile;
020    @FXML
021    AnchorPane tileTop;
022    @FXML
023    AnchorPane tileBottom;
024    @FXML
025    ToggleButton buttonSummary;
026    @FXML
027    ToggleButton buttonGraph;
028    @FXML
029    ToggleButton buttonMap;
030    @FXML
031    ToggleButton buttonTable;
032
033    public static VBox openTile;
034
035    ToggleGroup toggleGroup = new ToggleGroup();
036
037    public ViewSwitcher viewSwitcher = new ViewSwitcher(this);
038
039    /**
040     * Replaces the view displayed in the view holder with a new view.
041     *
042     * @param node
043     *            the view node to be swapped in.
044     */
045    public void setView(Node node) {
046        tileView.getChildren().setAll(node);
047    }
048
049    @FXML
050    private void initialize() {
051        tile.getChildren().remove(tileBottom); // So that the tiles are by
052                                               // default closed.
053
054        toggleGroup.getToggles().addAll(buttonGraph, buttonMap, buttonSummary,
055                buttonTable);
056        toggleGroup.selectToggle(buttonSummary);
057
058    }
059
060    @FXML
061    void loadSummary(ActionEvent event) {
062        toggleGroup.selectToggle(buttonSummary);
063        viewSwitcher.loadView(ViewSwitcher.SUMMARY);
064    }
065
066    @FXML
067    void loadMap(ActionEvent event) {
068        toggleGroup.selectToggle(buttonMap);
069        viewSwitcher.loadView(ViewSwitcher.MAP);
070    }
071
072    @FXML
073    void loadGraph(ActionEvent event) {
074        toggleGroup.selectToggle(buttonGraph);
075        viewSwitcher.loadView(ViewSwitcher.GRAPH);
076    }
077
078    @FXML
079    void loadTable(ActionEvent event) {
080        toggleGroup.selectToggle(buttonTable);
081        viewSwitcher.loadView(ViewSwitcher.TABLE);
082    }
083
084    @FXML
085    void hideContent(MouseEvent event) {
086        if (openTile == tile) {
087            tile.getChildren().remove(tileBottom);
088            openTile = null;
089        } else if (openTile == null) {
090            tile.getChildren().add(tileBottom);
091            openTile = tile;
092        } else {
093            openTile.getChildren().remove(1);
094            tile.getChildren().add(tileBottom);
095            openTile = tile;
096        }
097    }
098
099    @FXML
100    void hideContentButton(ActionEvent event) {
101        if (openTile == tile) {
102            tile.getChildren().remove(tileBottom);
103            openTile = null;
104        } else if (openTile == null) {
105            tile.getChildren().add(tileBottom);
106            openTile = tile;
107        } else {
108            openTile.getChildren().remove(1);
109            tile.getChildren().add(tileBottom);
110            openTile = tile;
111        }
112    }
113
114}
