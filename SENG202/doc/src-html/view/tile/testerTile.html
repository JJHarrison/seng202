001package view.tile;
002
003import javafx.application.Application;
004import javafx.scene.Scene;
005import javafx.scene.layout.VBox;
006import javafx.stage.Stage;
007
008public class testerTile extends Application {
009    public static void main(String[] args) {
010        launch(args);
011    }
012
013    @Override
014    public void start(Stage primaryStage) throws Exception {
015        VBox vBox = new VBox();
016        vBox.minHeight(600);
017        vBox.minWidth(1200);
018        vBox.getChildren().add(new Tile());
019        primaryStage.setScene(new Scene(vBox));
020        primaryStage.show();
021        primaryStage.setHeight(700);
022
023    }
024
025}
