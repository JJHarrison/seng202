001package view.user;
002
003import java.util.HashMap;
004import javafx.animation.KeyFrame;
005import javafx.animation.KeyValue;
006import javafx.animation.Timeline;
007import javafx.beans.property.DoubleProperty;
008import javafx.event.ActionEvent;
009import javafx.event.EventHandler;
010import javafx.fxml.FXMLLoader;
011import javafx.scene.Node;
012import javafx.scene.Parent;
013import javafx.scene.layout.StackPane;
014import javafx.util.Duration;
015
016/**
017 *
018 * @author Daniel van Wichen
019 */
020public class ScreenPane extends StackPane {
021        // Holds the screens to be displayed
022        private HashMap<String, Node> screens = new HashMap<>();
023
024        public ScreenPane() {
025                super();
026        }
027
028        // Add the screen to the collection
029        public void addScreen(String name, Node screen) {
030                screens.put(name, screen);
031        }
032
033        // Returns the Node with the appropriate name
034        public Node getScreen(String name) {
035                return screens.get(name);
036        }
037
038        /**
039         * Loads the FXML file, add the screen to the screens collection and
040         * finally injects the screenPane to the controller.
041         * @param name
042         * @param resource
043         * @return
044         */
045        public boolean loadScreen(String name, String resource) {
046                try {
047                        FXMLLoader myLoader = new FXMLLoader(getClass().getResource(resource));
048                        Parent loadScreen = (Parent) myLoader.load();
049                        ScreenController myScreenControler = ((ScreenController) myLoader.getController());
050                        myScreenControler.setScreenPane(this);
051                        addScreen(name, loadScreen);
052                        return true;
053                } catch (Exception e) {
054                        System.out.println(e.getMessage());
055                        return false;
056                }
057        }
058
059        // This method tries to displayed the screen with a predefined name.
060        // First it makes sure the screen has been already loaded. Then if there is
061        // more than
062        // one screen the new screen is been added second, and then the current
063        // screen is removed.
064        // If there isn't any screen being displayed, the new screen is just added
065        // to the root.
066
067        public boolean setScreen(final String name) {
068                if (screens.get(name) != null) { // screen loaded
069                        final DoubleProperty opacity = opacityProperty();
070
071                        if (!getChildren().isEmpty()) { // if there is more than one screen
072                                Timeline fade = new Timeline(new KeyFrame(Duration.ZERO,
073                                                new KeyValue(opacity, 1.0)), new KeyFrame(new Duration(
074                                                1000), new EventHandler<ActionEvent>() {
075                                        @Override
076                                        public void handle(ActionEvent t) {
077                                                getChildren().remove(0); // remove the displayed screen
078                                                getChildren().add(0, screens.get(name)); // add the
079                                                                                                                                        // screen
080                                                Timeline fadeIn = new Timeline(new KeyFrame(
081                                                                Duration.ZERO, new KeyValue(opacity, 0.0)),
082                                                                new KeyFrame(new Duration(1000), new KeyValue(
083                                                                                opacity, 1.0)));
084                                                fadeIn.play();
085                                        }
086                                }, new KeyValue(opacity, 0.0)));
087                                fade.play();
088
089                        } else {
090                                setOpacity(0.0);
091                                getChildren().add(screens.get(name));
092                                Timeline fadeIn = new Timeline(new KeyFrame(Duration.ZERO,
093                                                new KeyValue(opacity, 0.0)), new KeyFrame(new Duration(
094                                                1000), new KeyValue(opacity, 1.0)));
095                                fadeIn.play();
096                        }
097                        return true;
098                } else {
099                        System.out.println("screen hasn't been loaded!!! \n");
100                        return false;
101                }
102
103        }
104
105        // This method will remove the screen with the given name from the
106        // collection of screens
107        public boolean unloadScreen(String name) {
108                if (screens.remove(name) == null) {
109                        System.out.println("Screen didn't exist");
110                        return false;
111                } else {
112                        return true;
113                }
114        }
115
116}
