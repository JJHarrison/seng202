001package view.user;
002
003
004
005import javafx.event.ActionEvent;
006import javafx.event.EventHandler;
007import javafx.fxml.FXML;
008import javafx.scene.control.Button;
009import javafx.scene.control.ListView;
010import user.User;
011import data.persistant.Persistent;
012
013public class UserLoginController implements ScreenController {
014        @FXML
015        private ScreenPane myScreenPane;
016        
017        @FXML
018        Button buttonCreateProfile;
019        @FXML
020        Button buttonLogin;
021        @FXML
022        Button buttonCancel;
023        @FXML
024        ListView<User> userList;
025
026        @FXML
027        private void initialize() {
028
029                buttonCancel.setOnAction(new EventHandler<ActionEvent>() {
030
031                        public void handle(ActionEvent arg0) {
032                                System.exit(0);
033                        }
034                });
035
036                buttonLogin.setOnAction(new EventHandler<ActionEvent>() {
037
038                        public void handle(ActionEvent arg0) {
039                                System.out.println("Login");
040                        }
041                });
042
043                buttonCreateProfile.setOnAction(new EventHandler<ActionEvent>() {
044
045                        public void handle(ActionEvent arg0) {
046                                myScreenPane.setScreen("userCreate");
047                        }
048                });
049                
050                userList.setItems(Persistent.getUsers());
051                
052        }
053
054        @Override
055        public void setScreenPane(ScreenPane screenPage) {
056                myScreenPane = screenPage;
057                
058        }
059}
